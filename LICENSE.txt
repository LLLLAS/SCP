    __index = function(self, playerName)

-- Make bot invisible to other players (LocalTransparencyModifier)
utils.invisible = function()
    pcall(function()
        if not lplr.Character then return end
        for _, part in lplr.Character:GetDescendants() do
            if part:IsA('BasePart') then
                part.LocalTransparencyModifier = 1
            elseif part:IsA('Decal') or part:IsA('Texture') then
                part.Transparency = 1
            end
        end
    end)
end

utils.visible = function()
    pcall(function()
        if not lplr.Character then return end
        for _, part in lplr.Character:GetDescendants() do
            if part:IsA('BasePart') and part.Name ~= 'HumanoidRootPart' then
                part.LocalTransparencyModifier = 0
            elseif part:IsA('Decal') or part:IsA('Texture') then
                part.Transparency = 0
            end
        end
    end)
end

utils.purchase = function(type, itemname, last)
    if type == 'weapon' then
        local weapon = itemname:gsub('%s*%- %$%d+', '')
        if lplr.Character:FindFirstChild(weapon) or lplr.Backpack:FindFirstChild(weapon) then return 'owned' end
        local shopItem = workspace.Ignored.Shop:FindFirstChild(itemname)
        local click = shopItem and shopItem:FindFirstChildOfClass('ClickDetector')
        if not click then return 'nosuch' end
        local name = itemname:gsub('%[', ''):gsub('%]', ''):gsub('%s*%- %$%d+', '')
        local head = shopItem:FindFirstChildOfClass('Part') or shopItem:FindFirstChildOfClass('BasePart') or shopItem:FindFirstChild('Head')
        if not head then return 'nosuch' end
        local startingmoney = utils.getmoney()
        local saved = {}
        for i, v in lplr.Character:GetChildren() do
            if v:IsA('Tool') then table.insert(saved, v); v.Parent = lplr.Backpack end
        end
        local data = pguns[name:lower()]
        local part
        local check = false
        local timeout = tick()
        if data then
            for i, v in workspace.Ignored.Shop:GetChildren() do
                if data.name:lower() == v.Name:lower() then part = v; break end
            end
        end
        if not part then return 'nosuch' end
        task.spawn(function()
            repeat task.wait()
                check = startingmoney ~= utils.getmoney() or (lplr.Backpack:FindFirstChild(name) or (lplr.Character and lplr.Character:FindFirstChild(name)))
            until check or tick() - timeout > 0.9
        end)
        repeat task.wait()
            utils.update(head.CFrame + Vector3.new(0, 5, 0))
            if fireclickdetector then fireclickdetector(click) end
        until check or tick() - timeout > 0.90
        utils.update(CFrame.new(x(), yy(), x()))
        for i, v in saved do if v then v.Parent = lplr.Character end end
        local tool = lplr.Backpack:FindFirstChild(weapon) or lplr.Character:FindFirstChild(weapon)
        if tool then tool.Parent = lplr.Character; table.insert(guns, tool); return 'success' end
        return 'failed'
    elseif type == 'ammo' then
        local match = string.match(itemname, '%[([A-Za-z]+)%]')
        if not match then return 'nosuch' end
        local data = ammos[match:lower()]
        if not data then return 'nosuch' end
        local shopItem = workspace.Ignored.Shop:FindFirstChild(data.name)
        local clickdetector = shopItem and shopItem:FindFirstChildOfClass('ClickDetector')
        if not clickdetector then return 'nosuch' end
        local head = shopItem:FindFirstChildOfClass('Part') or shopItem:FindFirstChildOfClass('BasePart') or shopItem:FindFirstChild('Head')
        if not head then return 'nosuch' end
        local timeout = tick()
        local saved = {}
        for i, v in lplr.Character:GetChildren() do
            if v:IsA('Tool') then table.insert(saved, v); v.Parent = lplr.Backpack end
        end
        repeat task.wait()
            utils.update(head.CFrame + Vector3.new(0, 5, 0))
            if fireclickdetector then fireclickdetector(clickdetector) end
        until tick() - timeout > .75
        for i, v in saved do if v then v.Parent = lplr.Character end end
        if last then utils.update(CFrame.new(x(), yy(), x())) end
        return 'success'
    end
    return 'invalidtype'
end

local loadanimation = function(animation, speed)
    if not lplr or not lplr.Character or not lplr.Character:FindFirstChildOfClass('Humanoid') or not lplr.Character.Humanoid:FindFirstChild('Animator') then return end
    if currentanim then currentanim:Stop() end
    currentanim = nil
    local anim = Instance.new('Animation')
    local id = tostring(animation)
    if not id:match('^rbxassetid://') then id = 'rbxassetid://' .. id end
    anim.AnimationId = id
    local track = lplr.Character.Humanoid.Animator:LoadAnimation(anim)
    track.Priority = Enum.AnimationPriority.Action4
    track.Looped = true
    track:Play()
    track:AdjustSpeed(speed or 1)
    currentanim = track
    return track
end

-- COMBAT v141 — RenderStepped, stay on head, smart ammo (3 shots/gun/frame)
-- ══════════════════════════════════════════════════════════════════════════════
local _fire = replicated.MainEvent

runservice.RenderStepped:Connect(function()
    if not lplr or not lplr.Character or not lplr.Character:FindFirstChild('HumanoidRootPart') then return end
    local myRoot = lplr.Character.HumanoidRootPart
    
    -- ═══ KILLING ═══
    if killing and target and not buyinammo then
        local char = target.Character
        if not char then return end
        local head = char:FindFirstChild('Head')
        if not head then return end
        local hum = char:FindFirstChildOfClass('Humanoid')
        if not hum or hum.Health <= 0 then return end
        local be = char:FindFirstChild('BodyEffects')
        if be and be:FindFirstChild('K.O') and be['K.O'].Value then return end
        local myBE = lplr.Character:FindFirstChild('BodyEffects')
        if myBE and myBE:FindFirstChild('Reload') and myBE.Reload.Value then return end
        if char:FindFirstChildOfClass('ForceField') then return end
        
        local headPos = head.Position
        
        -- TP on top of target — STAY
        myRoot.CFrame = CFrame.new(headPos.X, headPos.Y + 7, headPos.Z)
        
        -- Buy guns if needed
        if #guns == 0 then
            for _, gn in ipairs(altcontrol.default_guns or {'rifle', 'flintlock'}) do
                if pguns[gn] then pcall(function() utils.purchase('weapon', pguns[gn].name) end) end
            end
            return
        end
        
        -- FIRE — smart ammo: shoot min(mag, 3) per gun per frame
        for i = 1, #guns do
            local v = guns[i]
            if not v then continue end
            local handle = v:FindFirstChild('Handle')
            if not handle then continue end
            if v.Parent ~= lplr.Character then v.Parent = lplr.Character end
            
            local mag = 999
            local ammoObj = v:FindFirstChild('Ammo')
            if ammoObj and ammoObj:IsA('ValueBase') then mag = ammoObj.Value end
            
            -- Empty → reload
            if mag <= 0 then
                _fire:FireServer('Reload', v)
                continue
            end
            
            -- Shoot max 10 per gun per frame
            local shots = math.min(mag, 10)
            if shots >= 1 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0,2,0), headPos, head, Vector3.zero) end
            if shots >= 2 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0,3,0), headPos, head, Vector3.zero) end
            if shots >= 3 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0,4,0), headPos, head, Vector3.zero) end
            if shots >= 4 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0,1,0), headPos, head, Vector3.zero) end
            if shots >= 5 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0,5,0), headPos, head, Vector3.zero) end
            if shots >= 6 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0.3,2,0), headPos, head, Vector3.zero) end
            if shots >= 7 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(-0.3,3,0), headPos, head, Vector3.zero) end
            if shots >= 8 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0,2,0.3), headPos, head, Vector3.zero) end
            if shots >= 9 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0,3,-0.3), headPos, head, Vector3.zero) end
            if shots >= 10 then _fire:FireServer('ShootGun', handle, headPos + Vector3.new(0,6,0), headPos, head, Vector3.zero) end
        end
        return
    end
    
    -- ═══ STOMPING ═══
    if stomping and target and target.Character then
        local lt = target.Character:FindFirstChild('LowerTorso')
        if lt then
            myRoot.CFrame = CFrame.new(lt.Position + Vector3.new(0, stomp_offset or 4, 0))
            myRoot.Velocity = Vector3.zero
            _fire:FireServer('Stomp')
        end
        return
    end
    
    -- ═══ IDLE ═══
    if showtarget then
        local sp = players:FindFirstChild(showtarget)
        if sp and sp.Character and sp.Character:FindFirstChild('HumanoidRootPart') then
            myRoot.CFrame = sp.Character.HumanoidRootPart.CFrame * CFrame.new(3, 0, 4)
            info.Text = 'following ' .. sp.Name
        end
    elseif void and not killing and not stomping then
        myRoot.CFrame = CFrame.new(x(), yy(), x())
        if info.Text ~= '' then info.Text = '' end
    end
end)

-- AUTO BUY AMMO — separate loop, checks every 5s, only when reserve = 0
-- ══════════════════════════════════════════════════════════════════════════════
task.spawn(function()
    local lastBuyTime = 0
    while true do
        task.wait(5)
        pcall(function()
            if buyinammo or not localent.Alive or #guns == 0 then return end
            
            -- Cooldown: don't buy again within 30s of last buy
            if tick() - lastBuyTime < 30 then return end
            
            -- Check if ANY gun has reserve = 0 AND magazine = 0
            local needBuy = false
            for _, v in guns do
                pcall(function()
                    if not v or not v:FindFirstChild('Ammo') then return end
                    local mag = v.Ammo:IsA('ValueBase') and v.Ammo.Value or 999
                    if mag > 0 then return end -- magazine has ammo, fine
                    
                    -- Magazine empty — check reserve
                    local reserve = 999
                    local inv = lplr:FindFirstChild('DataFolder') and lplr.DataFolder:FindFirstChild('Inventory')
                    if inv and inv:FindFirstChild(v.Name) then reserve = inv[v.Name].Value end
                    
                    if reserve <= 0 then needBuy = true end
                end)
            end
            
            if not needBuy then return end
            
            -- NEED TO BUY — pause killing, go to shop
            buyinammo = true
            local savedTarget = target
            local wasKilling = killing
            killing = false
            target = nil
            
            pcall(function() lplr.Character.HumanoidRootPart.CFrame = CFrame.new(x(), yy(), x()) end)
            info.Text = 'buying ammo...'
            
            for _, v in guns do
                pcall(function()
                    if not v then return end
                    -- Check if this gun needs ammo
                    local mag = 999
                    if v:FindFirstChild('Ammo') and v.Ammo:IsA('ValueBase') then mag = v.Ammo.Value end
                    local reserve = 999
                    local inv = lplr:FindFirstChild('DataFolder') and lplr.DataFolder:FindFirstChild('Inventory')
                    if inv and inv:FindFirstChild(v.Name) then reserve = inv[v.Name].Value end
                    if mag > 0 or reserve > 0 then return end -- still has ammo
                    
                    -- Find ammo in shop
                    local gunName = v.Name:lower():gsub('%[',''):gsub('%]',''):gsub('%s*%- %$%d+','')
                    for ammoKey, ammoData in pairs(ammos) do
                        if gunName:find(ammoKey) or ammoKey:find(gunName:sub(1,4)) then
                            local shopItem = workspace.Ignored.Shop:FindFirstChild(ammoData.name)
                            if not shopItem then break end
                            local click = shopItem:FindFirstChildOfClass('ClickDetector')
                            local shopHead = shopItem:FindFirstChildOfClass('Part') or shopItem:FindFirstChildOfClass('BasePart') or shopItem:FindFirstChild('Head')
                            if not click or not shopHead then break end
                            
                            -- Unequip tools
                            for _, tool in lplr.Character:GetChildren() do
                                if tool:IsA('Tool') then tool.Parent = lplr.Backpack end
                            end
                            task.wait(0.2)
                            
                            -- TP to shop + click
                            info.Text = 'buying: ' .. ammoKey
                            local timeout = tick()
                            repeat
                                pcall(function() lplr.Character.HumanoidRootPart.CFrame = shopHead.CFrame + Vector3.new(0, 3, 0) end)
                                if fireclickdetector then fireclickdetector(click) end
                                task.wait(0.15)
                            until tick() - timeout > 3
                            
                            -- Reload
                            pcall(function() _fire:FireServer('Reload', v) end)
                            task.wait(0.5)
                            break
                        end
                    end
                end)
            end
            
            -- Re-equip guns
            for _, v in guns do
                pcall(function() if v and v.Parent == lplr.Backpack then v.Parent = lplr.Character end end)
            end
            
            -- Resume
            if wasKilling and savedTarget and players:FindFirstChild(savedTarget.Name) and not bodyeffects.get(savedTarget, 'K.O') then
                target = savedTarget
                killing = true
                info.Text = 'resuming...'
            else
                void = true
                pcall(function() lplr.Character.HumanoidRootPart.CFrame = CFrame.new(x(), yy(), x()) end)
                info.Text = ''
            end
            buyinammo = false
            lastBuyTime = tick()
        end)
    end
end)

-- ANTI-STUCK: Auto-reset if killing/stomping stuck for too long
-- ══════════════════════════════════════════════════════════════════════════════
task.spawn(function()
    local killStartTime = 0
    local stompStartTime = 0
    local lastTargetHP = 0
    local lastHPDropTime = 0
    local buyinammoStart = 0
    local KILL_TIMEOUT = 12
    local STOMP_TIMEOUT = 8
    local NO_DAMAGE_TIMEOUT = 30
    local BUYAMMO_TIMEOUT = 10 -- buyinammo stuck max 10s
    
    while true do
        task.wait(1)
        pcall(function()
            -- buyinammo safety reset
            if buyinammo and buyinammoStart == 0 then
                buyinammoStart = tick()
            elseif not buyinammo then
                buyinammoStart = 0
            end
            if buyinammo and buyinammoStart > 0 and (tick() - buyinammoStart) > BUYAMMO_TIMEOUT then
                wsSend({type='log', msg='[anti-stuck] buyinammo stuck — resetting'})
                buyinammo = false
                buyinammoStart = 0
                void = true
            end
            
            -- Track when killing started
            if killing and killStartTime == 0 then
                killStartTime = tick()
                lastHPDropTime = tick()
                lastTargetHP = 999
            elseif not killing then
                killStartTime = 0
                lastHPDropTime = 0
                lastTargetHP = 0
            end
            
            if stomping and stompStartTime == 0 then
                stompStartTime = tick()
            elseif not stomping then
                stompStartTime = 0
            end
            
            -- Track target HP — detect if we're actually doing damage
            if killing and target and target.Character then
                local currentHP = 999
                pcall(function()
                    currentHP = target.Character.Humanoid.Health
                end)
                if currentHP < lastTargetHP then
                    -- HP dropped! We're doing damage
                    lastHPDropTime = tick()
                end
                lastTargetHP = currentHP
                
                -- NO DAMAGE TIMEOUT: if HP hasn't dropped in 30s, STOP
                if lastHPDropTime > 0 and (tick() - lastHPDropTime) > NO_DAMAGE_TIMEOUT then
                    local tname = target and target.Name or '?'
                    wsSend({type='log', msg='[anti-stuck] No damage on ' .. tname .. ' for 30s — stopping'})
                    killing = false
                    target = nil
                    void = true
                    killStartTime = 0
                    lastHPDropTime = 0
                    pcall(function() camera.CameraSubject = lplr.Character.Humanoid end)
                    pcall(function() utils.update(CFrame.new(x(), yy(), x())) end)
                    return
                end
            end
            
            -- Kill timeout
            if killing and killStartTime > 0 and (tick() - killStartTime) > KILL_TIMEOUT then
                local tname = target and target.Name or '?'
                wsSend({type='log', msg='[anti-stuck] Kill timeout on ' .. tname .. ' — resetting'})
                killing = false
                target = nil
                void = true
                killStartTime = 0
                pcall(function() camera.CameraSubject = lplr.Character.Humanoid end)
                pcall(function() utils.visible() end)
                pcall(function() utils.update(CFrame.new(x(), yy(), x())) end)
            end
            
            -- Stomp timeout
            if stomping and stompStartTime > 0 and (tick() - stompStartTime) > STOMP_TIMEOUT then
                local tname = target and target.Name or '?'
                wsSend({type='log', msg='[anti-stuck] Stomp timeout on ' .. tname .. ' — resetting'})
                stomping = false
                target = nil
                void = true
                stompStartTime = 0
                pcall(function() camera.CameraSubject = lplr.Character.Humanoid end)
                pcall(function() utils.visible() end)
                pcall(function() utils.update(CFrame.new(x(), yy(), x())) end)
            end
            
            -- Ghost state: killing=true but no target
            if killing and not target then
                wsSend({type='log', msg='[anti-stuck] Ghost kill state — resetting'})
                killing = false
                void = true
                pcall(function() utils.visible() end)
                pcall(function() utils.update(CFrame.new(x(), yy(), x())) end)
            end
            
            -- Ghost state: stomping=true but no target
            if stomping and not target then
                wsSend({type='log', msg='[anti-stuck] Ghost stomp state — resetting'})
                stomping = false
                void = true
                pcall(function() utils.visible() end)
                pcall(function() utils.update(CFrame.new(x(), yy(), x())) end)
            end
            
            -- Target left server
            if target and not players:FindFirstChild(target.Name) then
                wsSend({type='log', msg='[anti-stuck] Target left — resetting'})
                killing = false
                stomping = false
                target = nil
                void = true
                pcall(function() utils.visible() end)
                pcall(function() utils.update(CFrame.new(x(), yy(), x())) end)
            end
            
            -- Target has no character (respawning)
            if target and players:FindFirstChild(target.Name) and not target.Character then
                wsSend({type='log', msg='[anti-stuck] Target no character — resetting'})
                killing = false
                stomping = false
                target = nil
                void = true
                pcall(function() utils.visible() end)
                pcall(function() utils.update(CFrame.new(x(), yy(), x())) end)
            end
            
            -- Bot dead while killing
            if (killing or stomping) and not localent.Alive then
                killing = false
                stomping = false
                target = nil
                stats.deaths = stats.deaths + 1
            end
        end)
    end
end)

-- Bot protection: track who kills the bot and auto revenge
local revengeQueue = {} -- list of player names to revenge kill
local lastBotAttacker = nil -- updated inside the unified ClientBullet handler below

-- Auto reset on KO + revenge
task.spawn(function()
    repeat task.wait()
        if bodyeffects.get(lplr, 'K.O') then
            -- Find who's near the bot (stomper) or use last attacker
            local killer = nil
            pcall(function()
                if lplr.Character and lplr.Character:FindFirstChild('HumanoidRootPart') then
                    local botPos = lplr.Character.HumanoidRootPart.Position
                    local closest, closestDist = nil, 15
                    for _, v in players:GetPlayers() do
                        if v == lplr or table.find(altcontrol.owners, v.Name) or table.find(whitelist, v.Name) then continue end
                        if not v.Character or not v.Character:FindFirstChild('HumanoidRootPart') then continue end
                        local dist = (botPos - v.Character.HumanoidRootPart.Position).Magnitude
                        if dist < closestDist then closest = v; closestDist = dist end
                    end
                    if closest then killer = closest.Name end
                end
            end)
            -- Use last attacker if no one nearby
            if not killer and lastBotAttacker then killer = lastBotAttacker end

            -- Add to revenge queue
            if killer and not table.find(revengeQueue, killer) then
                table.insert(revengeQueue, killer)
                pcall(function() wsSend({type='log', msg='[revenge] ' .. killer .. ' killed bot — added to kill queue'}) end)
            end

            lastBotAttacker = nil
            pcall(function() lplr.Character.Humanoid.Health = 0 end)
            stats.deaths = stats.deaths + 1
            repeat task.wait() until localent.Alive
            utils.update(CFrame.new(x(), yy(), x()))

            -- Process revenge queue after respawn
            task.spawn(function()
                task.wait(1) -- wait for respawn
                while #revengeQueue > 0 and not stop do
                    local targetName = table.remove(revengeQueue, 1)
                    local user = utils.getname(targetName)
                    if user and user.Character and not bodyeffects.get(user, 'K.O') and not bodyeffects.get(user, 'SDeath') then
                        pcall(function() wsSend({type='log', msg='[revenge] Killing ' .. user.Name}) end)
                        target = user; killing = true
                        repeat task.wait() until not killing or stop
                        -- Auto stomp
                        if user and players:FindFirstChild(user.Name) and bodyeffects.get(user, 'K.O') then
                            stomping = true; target = user
                            repeat task.wait() until not stomping or stop
                        end
                        target = nil
                        utils.update(CFrame.new(x(), yy(), x()))
                        task.wait(0.5)
                    end
                end
            end)
        end
    until not game
end)

-- ══════════════════════════════════════════════════════════════════════════════
-- SENTRY + ASSIST — ClientBullet only (single source of truth)
-- ClientBullet fires BEFORE HealthChanged so reaction is instant and accurate.
-- We know exactly who shot whom → no "closest player" guessing.
-- ══════════════════════════════════════════════════════════════════════════════

-- Debounce: ignore the same shooter→victim pair within this window (seconds)
-- Prevents one spray volley from re-triggering the same kill 30 times.
local _sentryDebounce = {} -- [shooterName..victimName] = lastTick
local SENTRY_DEBOUNCE = 2.5

-- Cooldown per target for assist (owner can spray one enemy, bot picks up once)
local _assistDebounce = {} -- [victimName] = lastTick
local ASSIST_DEBOUNCE = 2.5

local function _tryKill(tgt, label)
    -- Only take over if bot is truly idle
    if killing or stomping or stop or buyinammo then return end
    if not tgt or not tgt.Character then return end
    if not tgt.Character:FindFirstChild('HumanoidRootPart') then return end
    if bodyeffects.get(tgt, 'K.O') or bodyeffects.get(tgt, 'SDeath') then return end
    if tgt.Character:FindFirstChildOfClass('ForceField') then return end
    pcall(function() wsSend({type='log', msg=label}) end)
    target = tgt; killing = true; void = false
end

-- Args: name, plr, handle, startPos, aimPos, hitPart, extra
replicated.MainEvent.OnClientEvent:Connect(function(name, plr, handle, startPos, aimPos, hitPart, ...)
    if name ~= 'ClientBullet' then return end
    if stop then return end

    -- Resolve shooter
    local shooter
    pcall(function()
        if typeof(plr) ~= 'Instance' then return end
        if plr:IsA('Player') then shooter = plr
        else shooter = players:GetPlayerFromCharacter(plr) or (plr.Parent and players:GetPlayerFromCharacter(plr.Parent)) end
    end)
    if not shooter or shooter == lplr then return end

    -- Track if the bot itself was hit (for revenge system)
    pcall(function()
        if typeof(hitPart) == 'Instance' and lplr.Character and hitPart:IsDescendantOf(lplr.Character) then
            if not table.find(altcontrol.owners, shooter.Name) and not table.find(whitelist, shooter.Name) then
                lastBotAttacker = shooter.Name
            end
        end
    end)

    -- Resolve victim from hitPart
    local victim
    pcall(function()
        if typeof(hitPart) ~= 'Instance' then return end
        for _, v in players:GetPlayers() do
            if v.Character and hitPart:IsDescendantOf(v.Character) then victim = v; break end
        end
    end)
    if not victim then return end

    -- ── SENTRY: someone shot an owner ──────────────────────────────────────
    if sentry and not killing and not stomping then
        local victimIsOwner = table.find(altcontrol.owners, victim.Name)
        local shooterIsOwner = table.find(altcontrol.owners, shooter.Name)
        local shooterIsWl = table.find(whitelist, shooter.Name)

        if victimIsOwner and not shooterIsOwner and not shooterIsWl then
            local key = shooter.Name .. victim.Name
            local now = tick()
            if not _sentryDebounce[key] or (now - _sentryDebounce[key]) > SENTRY_DEBOUNCE then
                _sentryDebounce[key] = now
                _tryKill(shooter, '[sentry] ' .. shooter.Name .. ' shot ' .. victim.Name .. ' — retaliating')
            end
        end
    end

    -- ── ASSIST: owner shot an enemy ────────────────────────────────────────
    if assist and not killing and not stomping then
        local shooterIsOwner = table.find(altcontrol.owners, shooter.Name)
        local victimIsOwner = table.find(altcontrol.owners, victim.Name)
        local victimIsWl = table.find(whitelist, victim.Name)
        local victimIsLplr = victim == lplr

        if shooterIsOwner and not victimIsOwner and not victimIsWl and not victimIsLplr then
            local now = tick()
            if not _assistDebounce[victim.Name] or (now - _assistDebounce[victim.Name]) > ASSIST_DEBOUNCE then
                _assistDebounce[victim.Name] = now
                _tryKill(victim, '[assist] ' .. shooter.Name .. ' shot ' .. victim.Name .. ' — helping')
            end
        end
    end
end)

-- ── Cleanup debounce tables on player leave (memory hygiene) ───────────────
players.PlayerRemoving:Connect(function(p)
    _sentryDebounce[p.Name .. p.Name] = nil -- edge case
    _assistDebounce[p.Name] = nil
    for k in pairs(_sentryDebounce) do
        if k:sub(1, #p.Name) == p.Name or k:sub(-#p.Name) == p.Name then
            _sentryDebounce[k] = nil
        end
    end
end)


-- ══════════════════════════════════════════════════════════════════════════════
-- MELEE ASSIST — Heartbeat HP-delta (covers punch, knife, stomp damage)
-- ClientBullet only fires for guns. For melee we detect HP drop + owner nearby.
-- PUNCH_RANGE = max studs owner can be from victim to count as "owner hit"
-- ══════════════════════════════════════════════════════════════════════════════
local PUNCH_RANGE = 10
local _meleeHP = {}

-- Seed initial HP
task.spawn(function()
    task.wait(1)
    for _, v in ipairs(players:GetPlayers()) do
        if v == lplr or table.find(altcontrol.owners, v.Name) or table.find(whitelist, v.Name) then continue end
        pcall(function()
            if v.Character and v.Character:FindFirstChildOfClass('Humanoid') then
                _meleeHP[v.Name] = v.Character.Humanoid.Health
            end
        end)
    end
end)

local function _seedMeleeHP(p)
    p.CharacterAdded:Connect(function(char)
        _meleeHP[p.Name] = nil
        task.wait(1)
        pcall(function() _meleeHP[p.Name] = char:FindFirstChildOfClass('Humanoid') and char.Humanoid.Health or nil end)
    end)
end
for _, p in ipairs(players:GetPlayers()) do _seedMeleeHP(p) end
players.PlayerAdded:Connect(_seedMeleeHP)
players.PlayerRemoving:Connect(function(p) _meleeHP[p.Name] = nil end)

runservice.Heartbeat:Connect(function()
    if not assist or killing or stomping or stop then return end
    pcall(function()
        for _, enemy in ipairs(players:GetPlayers()) do
            if enemy == lplr then continue end
            if table.find(altcontrol.owners, enemy.Name) then continue end
            if table.find(whitelist, enemy.Name) then continue end
            if not enemy.Character then continue end
            local hum = enemy.Character:FindFirstChildOfClass('Humanoid')
            if not hum or hum.Health <= 0 then continue end
            local be = enemy.Character:FindFirstChild('BodyEffects')
            if be and be:FindFirstChild('K.O') and be['K.O'].Value then continue end

            local currentHP = hum.Health
            local lastHP = _meleeHP[enemy.Name]
            if not lastHP then _meleeHP[enemy.Name] = currentHP; continue end

            if currentHP < lastHP then
                local er = enemy.Character:FindFirstChild('HumanoidRootPart')
                if er then
                    for _, ownerName in ipairs(altcontrol.owners) do
                        local owner = players:FindFirstChild(ownerName)
                        if not owner or not owner.Character then continue end
                        local or2 = owner.Character:FindFirstChild('HumanoidRootPart')
                        if not or2 then continue end
                        if (er.Position - or2.Position).Magnitude <= PUNCH_RANGE then
                            local now = tick()
                            if not _assistDebounce[enemy.Name] or (now - _assistDebounce[enemy.Name]) > ASSIST_DEBOUNCE then
                                _assistDebounce[enemy.Name] = now
                                _tryKill(enemy, '[assist-melee] ' .. ownerName .. ' hit ' .. enemy.Name .. ' — helping')
                            end
                            break
                        end
                    end
                end
            end
            _meleeHP[enemy.Name] = currentHP
        end
    end)
end)

-- Kill Queue Processor
task.spawn(function()
    while true do
        task.wait(0.2)
        if queueActive and #killQueue > 0 and not killing and not stomping and not stop then
            local nextTarget = table.remove(killQueue, 1)
            local user = utils.getname(nextTarget)
            if user then
                wsSend({type='log', msg='[queue] Killing ' .. user.Name .. ' (' .. #killQueue .. ' left)'})
                -- Kill
                killing = true
                target = user
                repeat task.wait() until not killing or stop
                if stop then queueActive = false break end
                stats.kills = stats.kills + 1
                -- No stomp — void back immediately
                void = true
                target = nil

        pcall(function()
            local pages = players:GetFriendsAsync(lplr.UserId)
            -- Try to find owner via API
            local userId
            local success, result = pcall(function()
                local url = 'https://users.roblox.com/v1/usernames/users'
                local body = http:JSONEncode({usernames = {ownerName}, excludeBannedUsers = false})
                local response = http:RequestAsync({Url = url, Method = 'POST', Headers = {['Content-Type'] = 'application/json'}, Body = body})
                local data = http:JSONDecode(response.Body)
                return data.data and data.data[1] and data.data[1].id
            end)
            if success and result then userId = result end
            if userId then
                local ok2, presence = pcall(function()
                    local url2 = 'https://presence.roblox.com/v1/presence/users'
                    local body2 = http:JSONEncode({userIds = {userId}})
                    local response2 = http:RequestAsync({Url = url2, Method = 'POST', Headers = {['Content-Type'] = 'application/json'}, Body = body2})
                    return http:JSONDecode(response2.Body)
                end)
                if ok2 and presence and presence.userPresences and presence.userPresences[1] then
                    local p = presence.userPresences[1]
                    if p.placeId and p.gameId then
                        wsSend({type='log', msg='Found! Joining ' .. ownerName .. ' server...'})
                        teleport:TeleportToPlaceInstance(p.placeId, p.gameId, lplr)
                        return
                    end
                end
            end
            wsSend({type='log', msg='Could not find ' .. ownerName .. ' server'})
        end)

    -- Server info/dashboard data
    elseif cmd == 'dashboard' then
        local playerCount = #players:GetPlayers()
        local money = utils.getmoney()
        local gunList = {}
        for _, g in guns do table.insert(gunList, g.Name) end
        wsSend({type='dashboard', data={
            username = lplr.Name,
            display = lplr.DisplayName,
            server = game.JobId,
            place = game.PlaceId,
            players = playerCount,
            money = money,
            guns = gunList,
            health = (function() local h = 0; pcall(function() h = math.floor((lplr.Character.Humanoid.Health / lplr.Character.Humanoid.MaxHealth) * 100) end); return h end)(),
            alive = localent.Alive,
            sentry = sentry, assist = assist, killaura = ka,
            void = void, automask = automask, autoarmor = autoarmor,
            autogun = autogun, farming = farming,
            karange = karange, stomp_offset = stomp_offset,
            stats = stats, queue = #killQueue, queueActive = queueActive,
        }})
    -- ══════════════════════════════════════════
    -- KILL QUEUE COMMANDS
    -- ══════════════════════════════════════════
    elseif cmd == 'queue' then
        -- Add single player to queue
        if targetName ~= '' then
            table.insert(killQueue, targetName)
            wsSend({type='log', msg='[queue] Added ' .. targetName .. ' | Queue: ' .. #killQueue})
        end
    elseif cmd == 'queueall' then
        -- Add ALL non-owner non-whitelist players to queue
        killQueue = {}
        for _, v in players:GetPlayers() do
            if v ~= lplr and not table.find(altcontrol.owners, v.Name) and not table.find(whitelist, v.Name) then
                table.insert(killQueue, v.Name)
            end
        end
        wsSend({type='log', msg='[queue] Added ALL (' .. #killQueue .. ' players)'})
    elseif cmd == 'queuestart' then
        if #killQueue == 0 then wsSend({type='log', msg='[queue] Empty!'}); return end
        queueActive = true
        wsSend({type='log', msg='[queue] Started! ' .. #killQueue .. ' targets'})
    elseif cmd == 'queuestop' then
        queueActive = false
        wsSend({type='log', msg='[queue] Stopped | ' .. #killQueue .. ' remaining'})
    elseif cmd == 'queueclear' then
        killQueue = {}
        queueActive = false
        wsSend({type='log', msg='[queue] Cleared'})
    elseif cmd == 'queuelist' then
        local list = #killQueue > 0 and table.concat(killQueue, ', ') or 'empty'
        wsSend({type='log', msg='[queue] ' .. #killQueue .. ': ' .. list})
    -- ══════════════════════════════════════════
    -- STATS COMMANDS
    -- ══════════════════════════════════════════
    elseif cmd == 'stats' then
        local uptime = math.floor(tick() - stats.startTime)
        local mins = math.floor(uptime / 60)
        wsSend({type='log', msg='[stats] K:' .. stats.kills .. ' S:' .. stats.stomps .. ' B:' .. stats.brings .. ' D:' .. stats.deaths .. ' | Up: ' .. mins .. 'min'})
    elseif cmd == 'statsreset' then
        stats = {kills=0, stomps=0, brings=0, deaths=0, startTime=tick()}
        wsSend({type='log', msg='[stats] Reset!'})
    end
end

-- WS Connection
local function connectWS()
    while true do
        local ok, socket = pcall(function()
            if WebSocket then return WebSocket.connect(WS_URL) end
            return nil
        end)
        if not ok or not socket then wsLog('WS failed. Retry 5s...'); task.wait(5); continue end
        wsSocket = socket; wsConnected = true; wsLog('Connected!')
        wsSend({ type = 'auth', token = WS_TOKEN, info = getStatus() })
        local statusThread = task.spawn(function()
            while wsConnected do pcall(function() wsSend({ type = 'status', data = getStatus() }) end); task.wait(2) end
        end)
        socket.OnMessage:Connect(function(raw)
            local ok2, msg = pcall(function() return http:JSONDecode(raw) end)
            if not ok2 then return end
            if msg.type == 'auth_ok' then wsLog('Auth OK!'); wsSend({type='log', msg='Bot: ' .. lplr.Name .. ' | ' .. game.JobId:sub(1,8)})
            elseif msg.type == 'auth_fail' then wsLog('Auth FAIL'); wsConnected = false
            elseif msg.type == 'command' then task.spawn(handleCommand, msg.cmd, msg.args) end
        end)
        socket.OnClose:Connect(function() wsLog('Disconnected'); wsConnected = false; wsSocket = nil end)
        repeat task.wait(1) until not wsConnected
        pcall(function() task.cancel(statusThread) end)
        wsLog('Reconnecting 5s...'); task.wait(5)
    end
end

loadanimation(default)
task.spawn(connectWS)
wsLog('Loaded! Player: ' .. lplr.Name)
